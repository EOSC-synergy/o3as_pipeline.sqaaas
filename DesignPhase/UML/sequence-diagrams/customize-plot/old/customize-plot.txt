@startuml
title "Customize Plot"

actor User
participant App
participant Graph
participant Sidebar
participant AppearanceSection
participant MaterialListBox as MDListBox #white
participant MaterialTextBox as MDTextBox #white
participant MaterialRangeSlider as MDRangeSlider #white

== Initialization ==
User -> Sidebar : OnClick()
activate Sidebar
Sidebar-> Sidebar : expand()
deactivate Sidebar
User -> AppearanceSection : OnClick()
activate AppearanceSection
AppearanceSection -> AppearanceSection : expand()
deactivate AppearanceSection

== Switch between different plot types ==
User -> MDListBox : OnClick()
activate MDListBox 
note over AppearanceSection : Wait for the User \n to select an item \n in the ListBox
AppearanceSection -> MDListBox : getType()
activate AppearanceSection
MDListBox ->> AppearanceSection : type
deactivate MDListBox 
AppearanceSection -> Graph : setType(type)
deactivate AppearanceSection
activate Graph
Graph -> App : update()
deactivate Graph

== Change title ==
User ->> MDTextBox : input
activate MDTextBox 
AppearanceSection -> MDTextBox : getInput()
activate AppearanceSection 
MDTextBox  ->> AppearanceSection: input
deactivate MDTextBox 
AppearanceSection -> Graph : setTitle(input)
deactivate AppearanceSection 
activate Graph
Graph -> App : update()
deactivate Graph

== Toggle range of x-axis and y-axis ==
User -> MDRangeSlider : toggle()
activate MDRangeSlider
AppearanceSection -> MDRangeSlider : getRange()
activate AppearanceSection
MDRangeSlider ->> AppearanceSection : range
deactivate MDRangeSlider
AppearanceSection -> Graph : setRange(range)
deactivate AppearanceSection
activate Graph
Graph -> App : update()
deactivate Graph
@enduml
